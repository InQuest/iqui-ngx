Basic Drop-down Button Element
Displays a button-styled drop-down

Usage 
    
      
      <iqui-dropdown-button
        [disabled="false"]
        [ngClass="{{'{}'}}"]
        [class=""]
        [size="{{attributes.size.slice(0, -1).join('|')}}"]
        [theme="{{attributes.theme.slice(0, -1).join('|')}}"]
        [position="{{attributes.position.slice(0, -1).join('|')}}"]
        [showOnFocus="{{attributes.showOnFocus.slice(0, -1).join('|')}}"]
        [showOnHover="{{attributes.showOnHover.slice(0, -1).join('|')}}"]
        [stayInViewport="{{attributes.stayInViewport.slice(0, -1).join('|')}}"]
        [toggleOnButtonClick="{{attributes.toggleOnButtonClick.slice(0, -1).join('|')}}"]
      >

        Drop-down Button content

        <ng-container *iquiDropdownHeader="let close">
          <!-- Using bootstrap's drop-down classes to style inside drop-down -->
          <span class="dropdown-header">
            Optional header content
            <a (click)="close()"> [X] <a>
          <span>
          <div class="dropdown-divider"><div>
        </ng-container>
        <ng-container *iquiDropdownBody="let close">
          <!-- Using bootstrap's drop-down classes to style inside drop-down -->
          <span class="dropdown-item"> Optional body item #1 <span>
          <span class="dropdown-item"> Optional body item #2 <span>
          <span class="dropdown-item"> Optional body item #3 <span>
        </ng-container>
        <ng-container *iquiDropdownFooter="let close">
          <!-- Using bootstrap's drop-down classes to style inside drop-down -->
          <div class="dropdown-divider"><div>
          <span class="dropdown-header">
            Optional footer content
            <a (click)="close()"> [X] <a>
          <span>
        </ng-container>

      </iqui-dropdown-button>
